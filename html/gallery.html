function shape(sides,length)
{
	a=360/sides;
	for(var n=0; n<sides; n++)
	{
		forward(length);
		left(a);
	}
}

function star(points,length) 
{ 
	a=360/points; 
	for(var n=0; n<points; n++) 
	{ 
		forward(length); 
		left(a*2); 
	} 
}



function spiral(length)
{
	if(length<30)
	{
		forward(length);
		left(15);
		spiral(length*1.02)
	}
}



function spiral_b()
{
	for( var i=1; i<216; i++ )
	{
		forward(i);
		left(70);
	}
}

function web()
{
	for( var i=0; i<36; i++ )
	{
		for( var j=0; j<36; j++ )
		{
			forward(16);
			left(10);
		}
		left(10);
	}
}



function random_walk(n) 
{ 
	if(n==undefined) { n=10000; }
	for( var i=0; i<n; i++ ) 
	{ 
		forward(5);
		left(Math.random()*360);
		if(x>width/2) { x=width/2; } 
		if(x<-width/2) { x=-width/2; } 
		if(y>height/2) { y=height/2; } 
		if(y<-height/2) { y=-height/2; } 
	} 
}

var hilbert_size;

function draw_hilbert(level)
{
	hilbert_size=192/Math.pow(2,level)
	hilbert(level,1)
}

function hilbert(level,parity)
{
	if( level<0 ) { return; }
	left(parity*90);
	hilbert(level-1,-parity);
	forward(hilbert_size);
	left(-parity*90);
	hilbert(level-1,parity);
	forward(hilbert_size);
	hilbert(level-1,parity);
	left(-parity*90)
	forward(hilbert_size)
	hilbert(level-1,-parity);
	left(parity*90)
}
